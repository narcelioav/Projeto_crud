var cadeira = {
cor: "Preta",
altura: "118",

};

document.write("Cor da cadeira: " + cadeira.cor + "br");
cadeira.peso = 17;
document.write("Peso da cadeira: " + cadeira.peso + "br");
document.write("Profundidade da cadeira: " + cadeira.profundidade + "br");
delete cadeira.profundidade;
document.write("Profundidade da cadeira: " + cadeira.profundidade + "br");
